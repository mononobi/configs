# execute this:

sh <(curl -sSf https://downloads.nordcdn.com/apps/linux/install.sh)

# add your user into nordvpn group then relogin to system:

sudo usermod -aG nordvpn USER_NAME

# login:

nordvpn login

# set autoconnect on:

nordvpn set autoconnect on

# set autoconnect off:

nordvpn set autoconnect off

# connect:

nordvpn connect de

# disconnect:

nordvpn disconnect
