# use one of these two commands for installation, no need to execute both of them (they will install the same thing):

sudo apt-get install transmission
sudo apt-get install transmission-gtk

# latest client (ubuntu <= 20.10):

sudo add-apt-repository ppa:transmissionbt/ppa
sudo apt-get update
sudo apt-get install transmission-gtk transmission-common

# latest client with remote and cli access (ubuntu <= 20.10):

sudo add-apt-repository ppa:transmissionbt/ppa
sudo apt-get update
sudo apt-get install transmission-gtk transmission-cli transmission-common transmission-daemon
