# if you don't need to install python2 on your system and also want to prevent some
# apps raising errors when they're accessing python. you can execute the 'script.sh'
# file and after that, apps which are accessing python will get access to
# python3 instead.

./script.sh

# now you can type python in the terminal and python3 shell will come up.

# NOTE: the script points to python3.9, but you can change it to point to any
# installed python version you'd like to.
